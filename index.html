Dữ liệu thông báo sẽ được lưu tự động lên Google Sheets

Mọi thành viên trong trường có thể xem và đồng bộ thông báo theo thời gian thực.

Bước 1 — Tạo Google Sheet
Tạo bảng tính với các cột tiêu đề

Mở sheets.google.com → Tạo sheet mới → Đặt tên tab là ThongBao → Thêm các cột hàng đầu tiên:

ID Tiêu đề Nội dung Mức độ Nhóm Người gửi Ngày gửi Hạn Đính kèm
Bước 2 — Tạo Apps Script Web App
Vào Extensions → Apps Script → Dán code sau
function doPost(e){var ss=SpreadsheetApp.getActiveSpreadsheet();var sh=ss.getSheetByName("ThongBao")||ss.getActiveSheet();var d=JSON.parse(e.postData.contents);sh.appendRow([d.id,d.title,d.content,d.priority,d.groups,d.sender,d.date,d.deadline,d.attachment]);return ContentService.createTextOutput(JSON.stringify({status:"ok"})).setMimeType(ContentService.MimeType.JSON);}
function doGet(){var ss=SpreadsheetApp.getActiveSpreadsheet();var sh=ss.getSheetByName("ThongBao")||ss.getActiveSheet();var rows=sh.getDataRange().getValues();return ContentService.createTextOutput(JSON.stringify(rows)).setMimeType(ContentService.MimeType.JSON);}
